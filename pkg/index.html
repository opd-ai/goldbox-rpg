Gold Box RPG JSON-RPC API Documentation

Overview

JSON-RPC 2.0 interface for the Gold Box RPG game server.

Connection Details

  • Base URL: http://localhost:8080/rpc
  • Protocol: HTTP/1.1
  • Content-Type: application/json
  • Method: POST

Base Request Format

{
    "jsonrpc": "2.0",
    "method": "Method.Name",
    "params": {},
    "id": 1
}

Methods

Combat.Start

Initiates combat sequence.

Parameters:

{
    "combatants": string[],
    "position": {
        "x": number,
        "y": number
    }
}

Examples:

// JavaScript
const response = await fetch('/rpc', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
        jsonrpc: '2.0',
        method: 'Combat.Start',
        params: {
            combatants: ['player_1', 'monster_1'],
            position: {x: 10, y: 10}
        },
        id: 1
    })
});

// Go
type CombatStartParams struct {
    Combatants []string    `json:"combatants"`
    Position   game.Position `json:"position"`
}

req := &JSONRPCRequest{
    JsonRPC: "2.0",
    Method:  "Combat.Start",
    Params:  CombatStartParams{
        Combatants: []string{"player_1", "monster_1"},
        Position:   game.Position{X: 10, Y: 10},
    },
    ID: 1,
}

# curl
curl -X POST http://localhost:8080/rpc \
  -H "Content-Type: application/json" \
  -d '{
    "jsonrpc": "2.0",
    "method": "Combat.Start",
    "params": {
        "combatants": ["player_1", "monster_1"],
        "position": {"x": 10, "y": 10}
    },
    "id": 1
  }'

Combat.Attack

Executes combat attack action.

Parameters:

{
    "actorId": string,
    "targetId": string,
    "weaponId": string
}

Examples:

// JavaScript
const response = await fetch('/rpc', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({
        jsonrpc: '2.0',
        method: 'Combat.Attack',
        params: {
            actorId: 'player_1',
            targetId: 'monster_1',
            weaponId: 'sword_1'
        },
        id: 1
    })
});

// Go
type CombatAttackParams struct {
    ActorID  string `json:"actorId"`
    TargetID string `json:"targetId"`
    WeaponID string `json:"weaponId"`
}

req := &JSONRPCRequest{
    JsonRPC: "2.0",
    Method:  "Combat.Attack",
    Params:  CombatAttackParams{
        ActorID:  "player_1",
        TargetID: "monster_1",
        WeaponID: "sword_1",
    },
    ID: 1,
}

# curl
curl -X POST http://localhost:8080/rpc \
  -H "Content-Type: application/json" \
  -d '{
    "jsonrpc": "2.0",
    "method": "Combat.Attack",
    "params": {
        "actorId": "player_1",
        "targetId": "monster_1",
        "weaponId": "sword_1"
    },
    "id": 1
  }'

Combat.EndTurn

Ends current combat turn.

Parameters:

{
    "actorId": string
}

Item.Drop

Creates item drop in game world.

Parameters:

{
    "itemId": string,
    "position": {
        "x": number,
        "y": number
    }
}
